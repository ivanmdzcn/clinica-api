-- MÓDULO DE PACIENTES (DATOS MAESTROS)
-- 1. Tabla principal: pacientes
create table pacientes (
    id int auto_increment primary key,
    dpi varchar(20) unique,
    nombres varchar(100) not null,
    apellidos varchar(100) not null,
    nombre_completo varchar(150) not null,
    fecha_nacimiento date not null,
    sexo enum('masculino', 'femenino', 'otro') not null,
    telefono varchar(20),
    direccion text,
    email varchar(100),
    fecha_registro timestamp default current_timestamp,
    creado_por int,
    foreign key (creado_por) references usuarios(id)
);

-- 2 Tabla relacionada: consultas (para registrar cada consulta médica realizada a un paciente)
CREATE TABLE consultas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    paciente_id INT NOT NULL,
    medico_id INT NOT NULL,
    
    tipo_consulta VARCHAR(50),  -- Enum ('primera_vez', 'seguimiento', 'emergencia') en el código de la aplicación
    motivo_consulta TEXT NOT NULL,
    diagnostico TEXT,
    evolucion_tratamiento TEXT,
    observaciones TEXT,
    proxima_cita DATE,
    
    fecha_consulta TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    estado VARCHAR(20) DEFAULT 'abierta',  -- En lugar de ENUM para mayor flexibilidad en el código de la aplicación
    
    FOREIGN KEY (paciente_id) REFERENCES pacientes(id) ON DELETE RESTRICT,
    FOREIGN KEY (medico_id) REFERENCES usuarios(id) ON DELETE RESTRICT,
    
    INDEX idx_paciente (paciente_id),
    INDEX idx_medico (medico_id),
    INDEX idx_fecha (fecha_consulta),
    INDEX idx_estado (estado)
);

-- 3. Tabla relacionada: paciente_alergias (para registrar alergias a medicamentos o elementos)
create table paciente_alergias (
    id int auto_increment primary key,
    paciente_id int not null,
    medicamento_o_elemento varchar(100) not null,
    reaccion text,
    foreign key (paciente_id) references pacientes(id) on delete cascade
);

-- 4. Tabla relacionada: paciente_antecedentes (para registrar antecedentes familiares y personales)
create table paciente_antecedentes (
    id int auto_increment primary key,
    paciente_id int not null,
    tipo enum('familiar', 'personal') not null,
    condicion varchar(150) not null,
    descripcion text,    
foreign key (paciente_id) references pacientes(id) on delete cascade
);
    
-- 5. Tabla relacionada: signos_vitales (para registrar los signos vitales tomados en cada consulta)
    CREATE TABLE signos_vitales (
    id INT AUTO_INCREMENT PRIMARY KEY,    
    consulta_id INT NOT NULL,
    enfermera_id INT NOT NULL,    
    presion_sistolica INT,
    presion_diastolica INT,
    temperatura DECIMAL(4,2),
    frecuencia_cardiaca INT,
    frecuencia_respiratoria INT,
    saturacion_oxigeno INT,    
    peso DECIMAL(6,2),         -- en kg
    altura DECIMAL(4,2),       -- en metros
    glucosa_capilar DECIMAL(5,2),
    nivel_dolor INT,           -- 0-10    
    observaciones TEXT,    
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,    
    FOREIGN KEY (consulta_id) REFERENCES consultas(id) ON DELETE CASCADE,
    FOREIGN KEY (enfermera_id) REFERENCES usuarios(id) ON DELETE RESTRICT
);

-- 6. Tabla relacionada: examen_fisico (para registrar los hallazgos del examen físico en cada consulta)
CREATE TABLE examen_fisico (
    id INT AUTO_INCREMENT PRIMARY KEY,
    consulta_id INT NOT NULL,
    es_normal BOOLEAN NOT NULL DEFAULT TRUE,
    descripcion TEXT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_examen_fisico_consulta
        FOREIGN KEY (consulta_id)
        REFERENCES consultas(id)
        ON DELETE CASCADE
);

-- 7. Tabla relacionada: diagnosticos (para registrar los diagnósticos realizados en cada consulta)
CREATE TABLE diagnosticos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    consulta_id INT NOT NULL,
    codigo_cie10 VARCHAR(10),
    descripcion VARCHAR(255) NOT NULL,
    tipo ENUM('presuntivo','definitivo') DEFAULT 'definitivo',
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_diagnostico_consulta
        FOREIGN KEY (consulta_id)
        REFERENCES consultas(id)
        ON DELETE CASCADE
);

-- 8. Tabla relacionada: tratamientos (para registrar los tratamientos indicados en cada consulta)
CREATE TABLE tratamientos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    consulta_id INT NOT NULL,
    descripcion TEXT NOT NULL,
    indicaciones TEXT,
    fecha_inicio DATE,
    fecha_fin DATE,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_tratamiento_consulta
        FOREIGN KEY (consulta_id)
        REFERENCES consultas(id)
        ON DELETE CASCADE
);

-- 9. Tabla: Orden de Laboratorio
CREATE TABLE orden_laboratorio (
    id INT PRIMARY KEY AUTO_INCREMENT,
    consulta_id INT NOT NULL,
    numero_orden VARCHAR(50) NOT NULL UNIQUE,
    fecha_orden DATE NOT NULL,
    paciente_id INT NOT NULL,
    medico_id INT NOT NULL,
    diagnostico_cie10 VARCHAR(10),
    observaciones TEXT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT fk_orden_laboratorio_consulta
        FOREIGN KEY (consulta_id)
        REFERENCES consultas(id)
        ON DELETE CASCADE,
    CONSTRAINT fk_orden_laboratorio_paciente
        FOREIGN KEY (paciente_id)
        REFERENCES pacientes(id)
        ON DELETE RESTRICT,
    CONSTRAINT fk_orden_laboratorio_medico
        FOREIGN KEY (medico_id)
        REFERENCES usuarios(id)
        ON DELETE RESTRICT
);

-- 10. Tabla: Exámenes de Laboratorio
CREATE TABLE examen_laboratorio (
    id_examen INT PRIMARY KEY AUTO_INCREMENT,
    id_orden INT NOT NULL,
    nombre_examen VARCHAR(100) NOT NULL,
    resultado VARCHAR(50),
    unidad VARCHAR(20),
    valor_referencia VARCHAR(50),
    fecha_resultado DATE,
    FOREIGN KEY (id_orden) REFERENCES orden_laboratorio(id_orden)
);

-- 11. Tabla relacionada: recetas (para registrar las recetas médicas indicadas en cada consulta)
CREATE TABLE recetas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    consulta_id INT NOT NULL,
    medicamento VARCHAR(150) NOT NULL,
    dosis VARCHAR(50),
    frecuencia VARCHAR(50),
    duracion VARCHAR(50),
    via_administracion VARCHAR(30),
    indicaciones TEXT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_receta_consulta
        FOREIGN KEY (consulta_id)
        REFERENCES consultas(id)
        ON DELETE CASCADE
);

CREATE TABLE configuracion_empresa (
    id INT AUTO_INCREMENT PRIMARY KEY,
    
    -- Datos básicos de identificación
    razon_social VARCHAR(200) NOT NULL,
    nombre_comercial VARCHAR(200),
    numero_identificacion_fiscal VARCHAR(50) UNIQUE,
    
    -- Dirección
    direccion TEXT NOT NULL,
    ciudad VARCHAR(100),
    estado_provincia VARCHAR(100),
    codigo_postal VARCHAR(20),
    pais VARCHAR(100) NOT NULL,
    
    -- Contacto
    telefono_principal VARCHAR(20),
    telefono_secundario VARCHAR(20),
    email_principal VARCHAR(100) NOT NULL,
    email_secundario VARCHAR(100),
    sitio_web VARCHAR(200),
    
    -- Configuración de Moneda
    codigo_moneda VARCHAR(3) NOT NULL DEFAULT 'GTQ',
    simbolo_moneda VARCHAR(10) NOT NULL DEFAULT 'Q',
    nombre_moneda VARCHAR(50) DEFAULT 'Quetzal',
    posicion_simbolo ENUM('antes', 'despues') DEFAULT 'antes',
    separador_miles VARCHAR(1) DEFAULT ',',
    separador_decimales VARCHAR(1) DEFAULT '.',
    decimales INT DEFAULT 2,
    
    -- Logo y marca
    logo_url VARCHAR(500),
    color_primario VARCHAR(7) DEFAULT '#3B82F6',
    
    -- Textos para documentos
    pie_pagina_documentos TEXT,
    mensaje_agradecimiento TEXT,
    
    -- Auditoría
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    actualizado_por INT,
    
    FOREIGN KEY (actualizado_por) REFERENCES usuarios(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Tabla de Médicos
CREATE TABLE medicos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    usuario_id INT NOT NULL,
    cedula_profesional VARCHAR(50) NOT NULL,
    especialidad VARCHAR(100) NOT NULL,
    subespecialidad VARCHAR(100) NULL,
    consultorio VARCHAR(50) NULL,
    telefono_consultorio VARCHAR(20) NULL,
    horario_atencion TEXT NULL,
    observaciones TEXT NULL,
    activo BOOLEAN NOT NULL DEFAULT TRUE,
    fecha_creacion DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT fk_medico_usuario FOREIGN KEY (usuario_id) 
        REFERENCES usuarios(id) ON DELETE CASCADE,
    CONSTRAINT uk_medico_usuario UNIQUE (usuario_id),
    CONSTRAINT uk_medico_cedula UNIQUE (cedula_profesional)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Índices para búsquedas
CREATE INDEX idx_medico_especialidad ON medicos(especialidad);
CREATE INDEX idx_medico_activo ON medicos(activo);

-- 7. MÓDULO DE MEDICINA PREVENTIVA (VACUNAS Y DESPARASITANTES)
create table inmunizaciones (
    id int auto_increment primary key,
    paciente_id int not null,
    consulta_id int,
    tipo enum('vacuna', 'desparasitante') not null,
    nombre_insumo varchar(100) not null,
    fecha_aplicacion date not null,
    lote varchar(50) not null,
    dosis_num int default 1,
    laboratorio varchar(100),
    proxima_dosis date,
    foreign key (paciente_id) references pacientes(id),
    foreign key (consulta_id) references consultas(id)
);
